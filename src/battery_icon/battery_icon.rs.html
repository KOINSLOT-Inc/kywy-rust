8use core::fmt::Write as Write2; // ‚Üê required for core::write! to work
9use defmt_rtt as _;
10use embassy_executor::Spawner;
11use embassy_time::{Duration, Timer};
12use embedded_graphics::{
13    mono_font::{MonoTextStyle, ascii::FONT_6X10},
14    pixelcolor::BinaryColor,
15    prelude::*,
16    text::Text,
17};
18use heapless::String;
19use kywy::battery::BatteryStatus;
20use kywy::{kywy_battery_from, kywy_display_from, kywy_spi_from, kywy_usb_from};
21use panic_probe as _;
22
23#[embassy_executor::main]
24async fn main(spawner: Spawner) {
25    let p = embassy_rp::init(Default::default());
26    kywy_spi_from!(p => spi_bus);
27    kywy_display_from!(spi_bus, p => display);
28    kywy_battery_from!(p => battery);
29    kywy_usb_from!(spawner, p);
30
31    display.initialize().await;
32    display.enable();
33
34    let text_style = MonoTextStyle::new(&FONT_6X10, BinaryColor::Off);
35
36    loop {
37        display.clear(BinaryColor::On).unwrap();
38
39        // Draw title
40        Text::new("Battery Status Test", Point::new(10, 40), text_style)
41            .draw(&mut display)
42            .unwrap();
43
44        // Read and display voltage
45        let voltage_mv = battery.read_voltage_mv().await;
46        let mut buf: String<32> = String::new();
47        write!(&mut buf, "Voltage: {} mV", voltage_mv).unwrap();
48
49        Text::new(&buf, Point::new(10, 60), text_style)
50            .draw(&mut display)
51            .unwrap();
52
53        let status = battery.status();
54        let mut status_buf: String<32> = String::new();
55        let status_str = match status {
56            BatteryStatus::Charging => "Charging",
57            BatteryStatus::Charged => "Charged",
58            BatteryStatus::NotCharging => "Not Charging",
59        };
60        core::write!(&mut status_buf, "Status: {}", status_str).unwrap();
61
62        Text::new(&status_buf, Point::new(10, 80), text_style)
63            .draw(&mut display)
64            .unwrap();
65
66        battery.draw_async(&mut display).await.unwrap();
67
68        display.write_display().await;
69        Timer::after(Duration::from_secs(1)).await;
70    }
71}
