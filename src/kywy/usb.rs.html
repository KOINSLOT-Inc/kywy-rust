20type MyUsbDriver = Driver<'static, USB>;

23static CONFIG_DESCRIPTOR: StaticCell<[u8; 256]> = StaticCell::new();
24static BOS_DESCRIPTOR: StaticCell<[u8; 256]> = StaticCell::new();
25static CONTROL_BUF: StaticCell<[u8; 64]> = StaticCell::new();
26static CDC_STATE: StaticCell<State> = StaticCell::new();

28// Channels for USB I/O
29static USB_RX_CHANNEL: Channel<CriticalSectionRawMutex, u8, 256> = Channel::new();
30static USB_TX_CHANNEL: Channel<CriticalSectionRawMutex, u8, 256> = Channel::new();

32bind_interrupts!(struct Irqs {
33    USBCTRL_IRQ => InterruptHandler<USB>;
34});

36#[embassy_executor::task]
37pub async fn usb_monitor_task(spawner: Spawner, usb: USB) {
38    let driver = Driver::new(usb, Irqs);
39
40    let config = {
41        let mut config = Config::new(0x2e8a, 0x00c0); // Arduino VID, Pico PID
42        config.manufacturer = Some("Arduino");
43        config.product = Some("RaspberryPi Pico");
44        config.serial_number = Some("A63064E62CA5324B"); // or None if you want dynamic serial
45        config.max_power = 250; // 500mA / 2, as per USB spec units
46        config.max_packet_size_0 = 64;
47        config.device_class = 0xef; // Miscellaneous Device
48        config.device_sub_class = 0x02; // Common Class
49        config.device_protocol = 0x01; // Interface Association Descriptor
50        config
51    };
52
53    let mut builder = Builder::new(
54        driver,
55        config,
56        CONFIG_DESCRIPTOR.init([0; 256]),
57        BOS_DESCRIPTOR.init([0; 256]),
58        &mut [],
59        CONTROL_BUF.init([0; 64]),
60    );
61
62    let cdc = CdcAcmClass::new(&mut builder, CDC_STATE.init(State::new()), 64);
63    let (mut sender, mut receiver, control) = cdc.split_with_control();
64
65    let usb = builder.build();
66    unwrap!(spawner.spawn(run_usb_task(usb)));
67
68    sender.wait_connection().await;
69    control.control_changed().await;
70
71    loop {
72        sender.wait_connection().await;
73        control.control_changed().await;
74
75        let dtr = sender.dtr();
76        let baud = receiver.line_coding().data_rate();
77        info!("DTR={}, baud={}", dtr, baud);
78
79        if dtr {
80            info!("Terminal connected. Starting echo...");
81            let _ = echo(&mut receiver, &mut sender).await;
82        }
83    }
84}

91#[derive(Debug)]
92struct Disconnected;
93
94impl From<EndpointError> for Disconnected {
95    fn from(val: EndpointError) -> Self {
96        match val {
97            EndpointError::BufferOverflow => panic!("USB buffer overflow"),
98            EndpointError::Disabled => Disconnected,
99        }
100    }
101}

103/// Full duplex USB communication: receive → RX queue, TX queue → send.
104async fn echo(
105    receiver: &mut Receiver<'static, MyUsbDriver>,
106    sender: &mut Sender<'static, MyUsbDriver>,
107) -> Result<(), Disconnected> {
108    let mut read_buf = [0; 64];
109    let mut write_buf = [0; 64];
110
111    loop {
112        let dtr = sender.dtr();
113        let baud = receiver.line_coding().data_rate();
114
115        if baud == 1200 && dtr {
116            Timer::after_millis(100).await;
117            reset_to_usb_boot(0, 0);
118        }
119
120        if !dtr {
121            return Ok(());
122        }
123
124        // Read USB → echo to terminal and enqueue for app
125        match receiver
126            .read_packet(&mut read_buf)
127            .with_timeout(Duration::from_millis(10))
128            .await
129        {
130            Ok(Ok(n)) if n > 0 => {
131                for &b in &read_buf[..n] {
132                    // 1. Send to application
133                    let _ = USB_RX_CHANNEL.try_send(b);
134                    // 2. Echo back to user
135                    let _ = USB_TX_CHANNEL.try_send(b);
136                }
137            }
138            _ => {}
139        }
140
141        // Flush queued app responses
142        let mut n = 0;
143        while n < write_buf.len() {
144            match USB_TX_CHANNEL.try_receive() {
145                Ok(b) => {
146                    write_buf[n] = b;
147                    n += 1;
148                }
149                Err(_) => break,
150            }
151        }
152
153        if n > 0 {
154            let _ = sender.write_packet(&write_buf[..n]).await;
155        }
156
157        Timer::after_millis(1).await;
158    }
159}

163/// Enqueue a byte slice for USB output.
164pub async fn usb_print(data: &[u8]) {
165    for &b in data {
166        let _ = USB_TX_CHANNEL.send(b).await;
167    }
168}
169
170/// Print a string followed by newline.
171pub async fn usb_println(line: &str) {
172    usb_print(line.as_bytes()).await;
173    usb_print(b"\r\n").await;
174}

176/// Read one character from the USB RX channel.

181/// Read a line into the given buffer, ending on `\r`, `\n`, or `\r\n`.
182pub async fn usb_read_line(buf: &mut [u8]) -> usize {
183    let mut i = 0;
184
185    while i < buf.len() {
186        let b = USB_RX_CHANNEL.receive().await;
187
188        // Treat \r or \n as end of line
189        if b == b'\r' || b == b'\n' {
190            // Optional: skip \n after \r to handle \r\n
191            if b == b'\r' {
192                if let Ok(b2) = USB_RX_CHANNEL.try_receive() {
193                    if b2 != b'\n' {
194                        // Not a \r\n combo, push b2 back manually if needed
195                        let _ = USB_RX_CHANNEL.try_send(b2);
196                    }
197                }
198            }
199            break;
200        }
201
202        buf[i] = b;
203        i += 1;
204    }
205
206    i
207}
