5//! src/button_poll.rs
6//! Button polling library for Kywy board (direct level check, no async)
7//! Returns true (Pressed) or false (Released)
8//! Requires feature 'button_poll'
9
10use embassy_rp::PeripheralRef;
11use embassy_rp::gpio::{Input, Level, Pull};
12use embassy_rp::peripherals::*;
13
14#[derive(Copy, Clone, Debug)]
15pub enum ButtonId {
16    Left,
17    Right,
18    DUp,
19    DDown,
20    DLeft,
21    DRight,
22    DCenter,
23}
24
25pub struct ButtonPins {
26    pub left: PeripheralRef<'static, PIN_12>,
27    pub right: PeripheralRef<'static, PIN_2>,
28    pub dup: PeripheralRef<'static, PIN_9>,
29    pub ddown: PeripheralRef<'static, PIN_3>,
30    pub dleft: PeripheralRef<'static, PIN_6>,
31    pub dright: PeripheralRef<'static, PIN_7>,
32    pub dcenter: PeripheralRef<'static, PIN_8>,
33}
34
35pub struct ButtonPoller {
36    left: Input<'static>,
37    right: Input<'static>,
38    dup: Input<'static>,
39    ddown: Input<'static>,
40    dleft: Input<'static>,
41    dright: Input<'static>,
42    dcenter: Input<'static>,
43}
44
45impl ButtonPoller {
46    pub fn new(pins: ButtonPins) -> Self {
47        fn mk_input<P: embassy_rp::gpio::Pin + 'static>(
48            pin: PeripheralRef<'static, P>,
49        ) -> Input<'static> {
50            let mut input = Input::new(pin, Pull::Up);
51            input.set_schmitt(true);
52            input
53        }
54
55        Self {
56            left: mk_input(pins.left),
57            right: mk_input(pins.right),
58            dup: mk_input(pins.dup),
59            ddown: mk_input(pins.ddown),
60            dleft: mk_input(pins.dleft),
61            dright: mk_input(pins.dright),
62            dcenter: mk_input(pins.dcenter),
63        }
64    }
65
66    pub fn is_pressed(&self, id: ButtonId) -> bool {
67        let level = match id {
68            ButtonId::Left => self.left.get_level(),
69            ButtonId::Right => self.right.get_level(),
70            ButtonId::DUp => self.dup.get_level(),
71            ButtonId::DDown => self.ddown.get_level(),
72            ButtonId::DLeft => self.dleft.get_level(),
73            ButtonId::DRight => self.dright.get_level(),
74            ButtonId::DCenter => self.dcenter.get_level(),
75        };
76
77        level == Level::Low
78    }
79
80    pub fn poll_all(&self) -> u8 {
81        let mut bits = 0u8;
82
83        if self.is_pressed(ButtonId::Left) {
84            bits |= 1 << 0;
85        }
86        if self.is_pressed(ButtonId::Right) {
87            bits |= 1 << 1;
88        }
89        if self.is_pressed(ButtonId::DUp) {
90            bits |= 1 << 2;
91        }
92        if self.is_pressed(ButtonId::DDown) {
93            bits |= 1 << 3;
94        }
95        if self.is_pressed(ButtonId::DLeft) {
96            bits |= 1 << 4;
97        }
98        if self.is_pressed(ButtonId::DRight) {
99            bits |= 1 << 5;
100        }
101        if self.is_pressed(ButtonId::DCenter) {
102            bits |= 1 << 6;
103        }
104
105        bits
106    }
107}
