9#[macro_export]
10macro_rules! kywy_spi_from {
11    ($peripherals:ident => $valname:ident) => {
12        let $valname = {
13            use embassy_rp::spi::{Async, Config, Spi};
14            use embassy_sync::blocking_mutex::raw::CriticalSectionRawMutex;
15            use embassy_sync::mutex::Mutex;
16            use static_cell::StaticCell;
17
18            // NOTE: we declare a STATIC cell with a different name than the return value
19            static SPI_BUS_STATIC: StaticCell<
20                Mutex<CriticalSectionRawMutex, Spi<'static, embassy_rp::peripherals::SPI0, Async>>,
21            > = StaticCell::new();
22
23            let spi = Spi::new(
24                $peripherals.SPI0,
25                $peripherals.PIN_18,
26                $peripherals.PIN_19,
27                $peripherals.PIN_16,
28                $peripherals.DMA_CH0,
29                $peripherals.DMA_CH1,
30                Config::default(),
31            );
32
33            let mutex = Mutex::new(spi);
34            let $valname = SPI_BUS_STATIC.init(mutex);
35            $valname
36        };
37    };
38}

40#[macro_export]
41macro_rules! kywy_display_from {
42    ($spi_bus:ident, $peripherals:ident => $var:ident) => {
43        use embassy_embedded_hal::shared_bus::asynch::spi::SpiDeviceWithConfig;
44        use embassy_rp::Peripheral;
45        use embassy_rp::gpio::{Level, Output};
46        use embassy_rp::spi::Config;
47        use embassy_rp::spi::Phase;
48        use embassy_rp::spi::Polarity;
49        use inverted_pin::InvertedPin;
50
51        //initialize pins
52        let cs_disp_pin = InvertedPin::new(Output::new($peripherals.PIN_17.into_ref(), Level::Low));
53        let disp_pin = Output::new($peripherals.PIN_22.into_ref(), Level::Low);
54
55        //initialize SPI device
56        let mut config = Config::default();
57        config.frequency = 1_000_000; // Try overclocking me! some displays can handle higher frequencies (~8_000_000), 2-4 MHz seems mostly stable but the spec says 1 MHz so leaving it there for now.
58        config.polarity = Polarity::IdleLow;
59        config.phase = Phase::CaptureOnFirstTransition;
60        let mut disp_spi = SpiDeviceWithConfig::new($spi_bus, cs_disp_pin, config);
61
62        //create display
63        let mut $var = $crate::display::KywyDisplay::new(
64            disp_spi, disp_pin, // DISP
65        );
66
67        //initialize display
68        $var.initialize().await;
69    };
70}

73#[macro_export]
74macro_rules! kywy_serial_usb_from {
75    ($peripherals:ident, $irqs:ident => $usb_var:ident, $serial_var:ident, $usb_fut:ident, $serial_fut:ident) => {{
76        use embassy_rp::peripherals::USB;
77        use embassy_rp::usb::Driver;
78        use embassy_usb::class::cdc_acm::CdcAcmClass;
79        use embassy_usb::class::cdc_acm::State;
80        use embassy_usb::{Builder, Config as USBConfig};
81        use log::info;
82
83        let mut config_descriptor = [0; 256];
84        let mut bos_descriptor = [0; 256];
85        let mut control_buf = [0; 64];
86        let mut state = State::new();
87
88        let driver = Driver::new($peripherals.USB, $irqs);
89
90        let mut config = USBConfig::new(0xc0de, 0xcafe);
91        config.manufacturer = Some("Kywy Devices");
92        config.product = Some("Kywy Serial");
93        config.serial_number = Some("KYWY-0001");
94        config.max_power = 100;
95        config.max_packet_size_0 = 64;
96
97        let mut builder = Builder::new(
98            driver,
99            config,
100            &mut config_descriptor,
101            &mut bos_descriptor,
102            &mut [],
103            &mut control_buf,
104        );
105
106        let mut $serial_var = CdcAcmClass::new(&mut builder, &mut state, 64);
107        let mut $usb_var = builder.build();
108
109        let mut buf = [0; 64];
110
111        let $usb_fut = $usb_var.run();
112
113        let $serial_fut = async {
114            loop {
115                $serial_var.wait_connection().await;
116                info!("Connected");
117                loop {
118                    match $serial_var.read_packet(&mut buf).await {
119                        Ok(n) if n > 0 => {
120                            let _ = $serial_var.write_packet(&buf[..n]).await;
121                        }
122                        _ => {
123                            info!("Disconnected");
124                            break;
125                        }
126                    }
127                }
128            }
129        };
130    }};
131}
