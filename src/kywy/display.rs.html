6//! Kywy Display Driver
7//! Compatable with embedded-graphics DrawTarget
8
9use core::ops::Not;
10use embassy_rp::gpio::Output;
11use embedded_graphics::{
12    Pixel,
13    draw_target::DrawTarget,
14    pixelcolor::BinaryColor,
15    prelude::{OriginDimensions, Size},
16};
17use embedded_hal_async::spi::SpiDevice;
18
19const WIDTH: usize = 144;
20const HEIGHT: usize = 168;
21const TOTAL_BUFFER_SIZE: usize = WIDTH * HEIGHT / 8;
22const BYTES_PER_LINE: usize = WIDTH / 8;
23const LINE_PACKET_SIZE: usize = 2 + BYTES_PER_LINE + 2;
24
25#[derive(Clone, Copy, PartialEq, Eq)]
26enum Vcom {
27    Lo = 0x00,
28    Hi = 0x40,
29}
30
31impl Not for Vcom {
32    type Output = Self;
33    fn not(self) -> Self::Output {
34        match self {
35            Vcom::Lo => Vcom::Hi,
36            Vcom::Hi => Vcom::Lo,
37        }
38    }
39}
40
41#[derive(Clone, Copy, PartialEq, Eq)]
42enum Command {
43    Nop = 0x00,
44    ClearMemory = 0x20,
45    WriteLine = 0x80,
46}
47
48pub struct KywyDisplay<'a, SPI> {
49    spi: SPI,
50    disp: Output<'a>,
51    buffer: [u8; TOTAL_BUFFER_SIZE],
52    line_buf: [u8; LINE_PACKET_SIZE],
53    vcom: Vcom,
54    auto_vcom: bool,
55}
56
57impl<'a, SPI> KywyDisplay<'a, SPI>
58where
59    SPI: SpiDevice,
60{
61    pub fn new(spi: SPI, disp: Output<'a>) -> Self {
62        Self {
63            spi,
64            disp,
65            buffer: [0x00; TOTAL_BUFFER_SIZE],
66            line_buf: [0x00; LINE_PACKET_SIZE],
67            vcom: Vcom::Hi,
68            auto_vcom: true, //defaults to toggling vcom every display update
69        }
70    }
71
72    pub async fn initialize(&mut self) {
73        self.disable();
74        self.vcom = Vcom::Hi;
75        self.clear_display().await;
76        self.enable();
77    }
78
79    pub fn enable(&mut self) {
80        self.disp.set_high();
81    }
82
83    pub fn disable(&mut self) {
84        self.disp.set_low();
85    }
86
87    pub async fn write_spi(&mut self, data: &[u8]) {
88        self.spi.transfer(&mut [], data).await.unwrap();
89    }
90
91    pub async fn write_display(&mut self) {
92        if self.auto_vcom {
93            self.vcom = !self.vcom;
94        }
95
96        for line in 0..HEIGHT {
97            self.line_buf[0] = Command::WriteLine as u8 | self.vcom as u8 | 0x80;
98            self.line_buf[1] = (line as u8 + 1).reverse_bits();
99
100            let buf_start = line * BYTES_PER_LINE;
101            for i in 0..BYTES_PER_LINE {
102                self.line_buf[2 + i] = self.buffer[buf_start + i].reverse_bits();
103            }
104
105            self.line_buf[2 + BYTES_PER_LINE] = 0x00;
106            self.line_buf[3 + BYTES_PER_LINE] = 0x00;
107
108            self.spi.write(&self.line_buf).await.unwrap();
109        }
110    }
111
112    pub async fn toggle_vcom(&mut self) {
113        self.vcom = !self.vcom;
114        self.write_spi(&[Command::Nop as u8 | self.vcom as u8, 0x00])
115            .await;
116    }
117
118    pub fn set_auto_vcom(&mut self, enable: bool) {
119        self.auto_vcom = enable;
120    }
121
122    pub fn is_auto_vcom(&self) -> bool {
123        self.auto_vcom
124    }
125
126    pub async fn clear_display(&mut self) {
127        if self.auto_vcom {
128            self.vcom = !self.vcom;
129        }
130        self.write_spi(&[Command::ClearMemory as u8 | self.vcom as u8, 0x00])
131            .await;
132    }
133
134    pub fn clear_buffer(&mut self, color: BinaryColor) {
135        let fill_value = if color.is_on() { 0xFF } else { 0x00 };
136        self.buffer.fill(fill_value);
137    }
138
139    pub fn set_pixel(&mut self, x: usize, y: usize, color: BinaryColor) {
140        if x >= WIDTH || y >= HEIGHT {
141            return;
142        }
143
144        let index = (y * BYTES_PER_LINE) + (x / 8);
145        let bit = x % 8;
146
147        if color.is_on() {
148            self.buffer[index] |= 1 << bit;
149        } else {
150            self.buffer[index] &= !(1 << bit);
151        }
152    }
153    pub fn height(&self) -> usize {
154        HEIGHT
155    }
156    pub fn width(&self) -> usize {
157        WIDTH
158    }
159}
160
161impl<'a, SPI> OriginDimensions for KywyDisplay<'a, SPI> {
162    fn size(&self) -> Size {
163        Size::new(WIDTH as u32, HEIGHT as u32)
164    }
165}
166
167impl<'a, SPI> DrawTarget for KywyDisplay<'a, SPI>
168where
169    SPI: SpiDevice,
170{
171    type Color = BinaryColor;
172    type Error = ();
173
174    fn draw_iter<I>(&mut self, pixels: I) -> Result<(), Self::Error>
175    where
176        I: IntoIterator<Item = Pixel<Self::Color>>,
177    {
178        for Pixel(coord, color) in pixels {
179            if coord.x >= 0 && coord.y >= 0 {
180                let x = coord.x as usize;
181                let y = coord.y as usize;
182                if x < WIDTH && y < HEIGHT {
183                    self.set_pixel(x, y, color);
184                }
185            }
186        }
187        Ok(())
188    }
189}
