5//! examples/menu.rs: Example of building menus with the Kywy device.
6//! This example uses the embedded_menu library to display a simple dummy menu
7//! Compile with: cargo build --example button_test_async --target thumbv6m-none-eabi --release
8
9#![no_std]
10#![no_main]
11
12use defmt::*;
13use defmt_rtt as _;
14use panic_probe as _;
15
16use kywy::button_async::{ButtonEvent, ButtonId, ButtonState};
17use kywy::{kywy_button_async_from, kywy_display_from, kywy_spi_from, kywy_usb_from}; // Import the macros
18
19use embassy_executor::Spawner;
20
21use embedded_graphics::Drawable;
22use embedded_graphics::{
23    mono_font::{MonoTextStyle, ascii::FONT_6X10},
24    pixelcolor::BinaryColor,
25    prelude::*,
26    text::Text,
27};
28use embedded_menu::{
29    Menu, MenuStyle,
30    interaction::programmed::Programmed,
31    interaction::{Action, Interaction, Navigation},
32    selection_indicator::style::Triangle,
33};
34
35#[embassy_executor::main]
36async fn main(spawner: Spawner) {
37    info!("Button test example started");
38    let p = embassy_rp::init(Default::default());
39
40    // Initialize display
41    info!("Initializing display");
42    kywy_spi_from!(p => spi_bus);
43    kywy_display_from!(spi_bus, p => display);
44    kywy_usb_from!(spawner, p);
45    display.initialize().await;
46    display.enable();
47    display.clear_buffer(BinaryColor::On);
48    let style = MonoTextStyle::new(&FONT_6X10, BinaryColor::Off);
49    write_message(&mut display, "Display Initialized", style);
50
51    // Initialize buttons
52    kywy_button_async_from!(&spawner, p => button_channel);
53
54    // initialize menu with theme
55    let menu_style = MenuStyle::new(BinaryColor::Off) // color should be off (black), the default is on and that is just white on white, that one really threw me for a loop there
56        .with_font(&FONT_6X10) // set font size
57        .with_selection_indicator(Triangle) // set indicator type
58        .with_input_adapter(Programmed); // set input adapter type, Programmed is what we want for mapping button inputs
59
60    let mut menu = Menu::with_style("Main Menu", menu_style) // here we define the actual menu layout, these dont actually do anything
61        .add_item("Start", ">", |_| ()) // add item to menu, the first string is the name of the item, the second is the indicator, and the third is a closure that is called when the item is selected
62        .add_item("Settings", false, |_| ()) // add item to menu, the first string is the name of the item, the second is the indicator, and the third is a closure that is called when the item is selected but this time with a boolean indicator
63        .add_item("About", false, |_| ()) // same as last but says about, none of these actually do anything
64        .build();
65    display.clear_buffer(BinaryColor::On); // clear perevious message
66    menu.update(&display); // update the menu with the init state
67    menu.draw(&mut display).unwrap(); // write to display buffer
68    display.write_display().await; // update display
69
70    loop {
71        let event: ButtonEvent = button_channel.receive().await; // wait for button event
72        let interaction = match (event.id, event.state) {
73            // match button event to desired input and map to a menu interaction
74            (ButtonId::DUp, ButtonState::Pressed) => {
75                Some(Interaction::Navigation(Navigation::Previous))
76            }
77            (ButtonId::DDown, ButtonState::Pressed) => {
78                Some(Interaction::Navigation(Navigation::Next))
79            }
80            (ButtonId::Right, ButtonState::Pressed) => Some(Interaction::Action(Action::Select)),
81            _ => None,
82        };
83        if let Some(i) = interaction {
84            display.clear_buffer(BinaryColor::On);
85            menu.interact(i); // sends the interaction to the menu system
86            menu.update(&display); // update the menu with the new state
87            menu.draw(&mut display).unwrap(); // draws the menu to display buffer
88            display.write_display().await; // update display
89        }
90    }
91}

93fn write_message<D: DrawTarget<Color = BinaryColor>>(
94    // this function is used to write a string to the display
95    display: &mut D,
96    msg: &str,
97    style: MonoTextStyle<BinaryColor>,
98) {
99    display.clear(BinaryColor::On).ok(); // clear screen
100    let _ = Text::new(msg, Point::new(5, 20), style).draw(display);
101}
