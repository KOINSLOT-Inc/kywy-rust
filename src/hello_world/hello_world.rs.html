5//! Hello World example for Kywy display
6
7#![no_std]
8#![no_main]
9use defmt::*;
10use defmt_rtt as _;
11use embassy_executor;
12use embassy_time::{Duration, Timer};
13use embedded_graphics::{
14    mono_font::{MonoTextStyle, ascii::FONT_6X10},
15    pixelcolor::BinaryColor,
16    prelude::*,
17    text::Text,
18};
19use kywy::kywy_display_from;
20use kywy::kywy_spi_from;
21use kywy::kywy_usb_from;
22use panic_probe as _;
23
24#[embassy_executor::main]
25async fn main(spawner: embassy_executor::Spawner) {
26    info!("Starting Kywy Hello World display!");
27
28    let p = embassy_rp::init(Default::default());
29    kywy_spi_from!(p => spi_bus);
30    kywy_display_from!(spi_bus, p => display);
31    kywy_usb_from!(spawner, p);
32
33    // Black Screen
34    display.clear_buffer(BinaryColor::Off); // Clear buffer to black
35    display.write_display().await;
36    Timer::after(Duration::from_millis(1000)).await;
37
38    // White Screen
39    display.clear_buffer(BinaryColor::On); // Clear buffer to white
40    display.write_display().await;
41    Timer::after(Duration::from_millis(1000)).await;
42
43    // Checkerboard pattern
44    for y in 0..display.size().height as usize {
45        for x in 0..display.size().width as usize {
46            let on = (x + y) % 2 == 0;
47            display.set_pixel(
48                x,
49                y,
50                if on {
51                    BinaryColor::On
52                } else {
53                    BinaryColor::Off
54                },
55            );
56        }
57    }
58    display.write_display().await;
59    Timer::after(Duration::from_millis(1000)).await;
60
61    // Write Hello, Rust! to display using the embedded-graphics crate
62    display.clear_buffer(BinaryColor::On); // Clear to white background
63    let style = MonoTextStyle::new(&FONT_6X10, BinaryColor::Off); // Black text
64
65    Text::new("Hello, Rust!", Point::new(10, 20), style)
66        .draw(&mut display)
67        .unwrap();
68
69    display.write_display().await;
70
71    // Main loop - toggle VCOM every second to prevent DC bias
72    loop {
73        display.toggle_vcom().await;
74        Timer::after(Duration::from_secs(1)).await;
75    }
76}
