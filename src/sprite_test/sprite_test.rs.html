4//! This example demonstrates the use of the `Sprite` struct from the `kywy` crate.
5//! It creates a sprite from a sprite sheet and animates it on a display.
6//! This only uses sprites.rs and no additional engine components to create animations with button events and move a sprite around with the dpad.
7
8#![no_std]
9#![no_main]
10
11use defmt::*;
12use defmt_rtt as _;
13use embassy_executor::Spawner;
14use embassy_time::{Duration, Timer};
15use embedded_graphics::{pixelcolor::BinaryColor, prelude::*};
16use heapless::Vec;
17use kywy::{
18    button_async::{ButtonId, ButtonState},
19    engine::sprite::{Animation, SpriteInstance, SpriteSheet},
20    kywy_button_async_from, kywy_display_from, kywy_spi_from, kywy_usb_from,
21};
22use panic_probe as _;
23
24#[embassy_executor::main]
25async fn main(spawner: Spawner) {
26    info!("Sprite D-Pad control test");
27
28    let p = embassy_rp::init(Default::default());
29    kywy_spi_from!(p => spi_bus);
30    kywy_display_from!(spi_bus, p => display);
31    kywy_button_async_from!(&spawner, p => button_channel);
32    kywy_usb_from!(spawner, p);
33
34    display.initialize().await;
35    display.enable();
36
37    static SPRITE_DATA: &[u8] = include_bytes!("../examples/Art Assets/monsters/electric.bmp");
38    let sheet = SpriteSheet::new(SPRITE_DATA, Size::new(64, 64)).unwrap();
39
40    let idle: &[(u32, u32)] = &[(0, 0), (1, 0), (2, 0)];
41    let left_trigger: &[(u32, u32)] = &[(0, 1), (1, 1), (2, 1)];
42    let right_trigger: &[(u32, u32)] = &[(0, 2), (1, 2), (2, 2)];
43
44    let mut animations: Vec<_, 4> = Vec::new();
45    animations.push(Animation::new(&sheet, idle, true)).unwrap(); // 0
46    animations
47        .push(Animation::new(&sheet, left_trigger, false))
48        .unwrap(); // 1
49    animations
50        .push(Animation::new(&sheet, right_trigger, false))
51        .unwrap(); // 2
52
53    let mut sprite = SpriteInstance::new(animations, Point::new(40, 40));
54
55    loop {
56        // Draw current frame
57        let frame = sprite.current().current_frame_sprite().unwrap();
58        display.clear(BinaryColor::On).unwrap();
59        frame.draw(&mut display, sprite.position).unwrap();
60        display.write_display().await;
61
62        // Animate and revert if needed
63        sprite.update(0); // 0 = idle
64
65        // Input handling
66        if let Ok(event) = button_channel.try_receive() {
67            if event.state == ButtonState::Pressed {
68                match event.id {
69                    ButtonId::Left => sprite.trigger(1),
70                    ButtonId::Right => sprite.trigger(2),
71                    ButtonId::DLeft => sprite.move_by(-5, 0),
72                    ButtonId::DRight => sprite.move_by(5, 0),
73                    ButtonId::DUp => sprite.move_by(0, -5),
74                    ButtonId::DDown => sprite.move_by(0, 5),
75                    _ => {}
76                }
77            }
78        }
79
80        Timer::after(Duration::from_millis(100)).await;
81    }
82}
