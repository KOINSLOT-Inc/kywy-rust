5//! examples/button_test.rs: Example to test button library on the Kywy board.
6//! Probably dont actually update the display on every button press, but this is just a test and I wanted to show that the commands are all processed.
7//!
8//! Compile with: cargo build --example button_test_async --target thumbv6m-none-eabi --release --features button-async --no-default-features
9
10#![no_std]
11#![no_main]
12
13use defmt::*;
14use defmt_rtt as _;
15use panic_probe as _;
16
17use kywy::button_async::{ButtonEvent, ButtonId, ButtonState};
18use kywy::{kywy_button_async_from, kywy_display_from, kywy_spi_from, kywy_usb_from}; // Import the macros
19
20use embassy_executor::Spawner;
21use embassy_time::{Duration, Timer};
22
23use embedded_graphics::{
24    mono_font::{MonoTextStyle, ascii::FONT_6X10},
25    pixelcolor::BinaryColor,
26    prelude::*,
27    text::Text,
28};
29
30#[embassy_executor::main]
31async fn main(spawner: Spawner) {
32    info!("Button test example started");
33    let p = embassy_rp::init(Default::default());
34
35    // Initialize display
36    info!("Initializing display");
37    kywy_spi_from!(p => spi_bus);
38    kywy_display_from!(spi_bus, p => display);
39    kywy_usb_from!(spawner, p);
40    let style = MonoTextStyle::new(&FONT_6X10, BinaryColor::Off);
41    display.initialize().await;
42    display.enable();
43    display.clear_buffer(BinaryColor::On);
44    write_message(&mut display, "Button Test Started", style);
45    display.write_display().await;
46
47    // Initialize buttons
48    info!("Initializing buttons");
49    kywy_button_async_from!(&spawner, p => button_channel);
50    info!("Buttons initialized");
51
52    display.clear(BinaryColor::Off).ok();
53    write_message(&mut display, "Buttons Initialized", style);
54    display.write_display().await;
55
56    loop {
57        let event: ButtonEvent = button_channel.receive().await;
58
59        let msg = match (event.id, event.state) {
60            (ButtonId::Left, ButtonState::Pressed) => "Left Pressed",
61            (ButtonId::Left, ButtonState::Released) => "Left Released",
62            (ButtonId::Right, ButtonState::Pressed) => "Right Pressed",
63            (ButtonId::Right, ButtonState::Released) => "Right Released",
64            (ButtonId::DUp, ButtonState::Pressed) => "D-Up Pressed",
65            (ButtonId::DUp, ButtonState::Released) => "D-Up Released",
66            (ButtonId::DDown, ButtonState::Pressed) => "D-Down Pressed",
67            (ButtonId::DDown, ButtonState::Released) => "D-Down Released",
68            (ButtonId::DLeft, ButtonState::Pressed) => "D-Left Pressed",
69            (ButtonId::DLeft, ButtonState::Released) => "D-Left Released",
70            (ButtonId::DRight, ButtonState::Pressed) => "D-Right Pressed",
71            (ButtonId::DRight, ButtonState::Released) => "D-Right Released",
72            (ButtonId::DCenter, ButtonState::Pressed) => "D-Center Pressed",
73            (ButtonId::DCenter, ButtonState::Released) => "D-Center Released",
74        };
75
76        info!("{}", msg);
77        write_message(&mut display, msg, style);
78        display.write_display().await;
79
80        Timer::after(Duration::from_millis(100)).await;
81    }
82}
83
84fn write_message<D: DrawTarget<Color = BinaryColor>>(
85    display: &mut D,
86    msg: &str,
87    style: MonoTextStyle<BinaryColor>,
88) {
89    display.clear(BinaryColor::On).ok(); // clear screen
90    let _ = Text::new(msg, Point::new(5, 20), style).draw(display);
91}
