5//! examples/usb_serial.rs: Example to test serial communication on the Kywy board.
6
7#![no_std]
8#![no_main]
9
10use defmt::*;
11use defmt_rtt as _;
12use panic_probe as _;
13
14use kywy::kywy_usb_from;
15use kywy::usb::{usb_println, usb_read_line};
16
17use core::fmt::Write;
18use embassy_executor::Spawner;
19use embassy_time::Timer;
20use heapless::String; // Needed for heapless::String
21
22#[embassy_executor::main]
23async fn main(spawner: Spawner) {
24    info!("usb_serial.rs started");
25
26    let p = embassy_rp::init(Default::default());
27    kywy_usb_from!(spawner, p);
28
29    Timer::after_millis(500).await; // wait for USB to initialize
30    loop {
31        usb_println("Enter your name:").await;
32
33        let mut buf = [0u8; 64]; // buffer to store data
34        let len = usb_read_line(&mut buf).await; // read line from USB
35
36        let name = core::str::from_utf8(&buf[..len])
37            .unwrap_or("stranger")
38            .trim_end();
39
40        let mut line: String<64> = String::new(); // initialize heapless::String
41        let _ = core::write!(line, "Hello, {}!", name); // write formatted string to heapless::String
42
43        usb_println(&line).await; // print hello message
44    }
45}
